import clr
import sys
import System
from System.IO import Path, Directory

# 1. Import Revit API & Services
clr.AddReference('RevitAPI')
from Autodesk.Revit.DB import *

clr.AddReference("RevitServices")
import RevitServices
from RevitServices.Persistence import DocumentManager

# 2. Import Windows Forms สำหรับ UI เลือกไฟล์และโฟลเดอร์
clr.AddReference('System.Windows.Forms')
from System.Windows.Forms import OpenFileDialog, FolderBrowserDialog, DialogResult

# 3. Import Excel Interop
clr.AddReference("Microsoft.Office.Interop.Excel")
import Microsoft.Office.Interop.Excel as Excel

app = DocumentManager.Instance.CurrentUIApplication.Application

# --- ส่วนที่ 1: UI เลือกไฟล์ Excel และโฟลเดอร์ Revit ---
excel_path = ""
dialog = OpenFileDialog()
dialog.Filter = "Excel files (*.xlsx)|*.xlsx|All files (*.*)|*.*"
dialog.Title = "1. กรุณาเลือกไฟล์ Excel สำหรับข้อมูลผู้ออกแบบ"

if dialog.ShowDialog() == DialogResult.OK:
    excel_path = dialog.FileName
else:
    OUT = "ยกเลิกการเลือกไฟล์ Excel"
    sys.exit()

rvt_folder = ""
folder_dialog = FolderBrowserDialog()
folder_dialog.Description = "2. กรุณาเลือกโฟลเดอร์ที่เก็บไฟล์ Revit"

if folder_dialog.ShowDialog() == DialogResult.OK:
    rvt_folder = folder_dialog.SelectedPath
else:
    OUT = "ยกเลิกการเลือกโฟลเดอร์ Revit"
    sys.exit()

# --- ส่วนที่ 2: อ่านข้อมูลจาก Excel แค่ครั้งเดียว ---
excel_app = Excel.ApplicationClass()
excel_app.Visible = False 
excel_data = []

try:
    workbook = excel_app.Workbooks.Open(excel_path)
    worksheet = workbook.Worksheets[1]
    
    # อ่านถึงแถวที่ 20 (ปรับเปลี่ยนได้ตามการใช้งานจริง)
    for row in range(4, 21):
        param_name = worksheet.Cells[row, 1].Value2
        thai_name = worksheet.Cells[row, 2].Value2
        license_id = worksheet.Cells[row, 4].Value2
        
        if param_name:
            excel_data.append({
                "param_name": str(param_name),
                "thai_name": str(thai_name) if thai_name else "",
                "license_id": str(license_id) if license_id else ""
            })
    workbook.Close(False)
except Exception as ex:
    excel_app.Quit()
    OUT = "เกิดข้อผิดพลาดในการอ่าน Excel: " + str(ex)
    sys.exit()
finally:
    excel_app.Quit()

# --- ส่วนที่ 3: ตรวจสอบประเภทไฟล์, เปิด, แก้ไข, เซฟ/Sync และ ปิด ---
results = []
local_save_dir = System.Environment.GetFolderPath(System.Environment.SpecialFolder.MyDocuments)
username = app.Username

rvt_files = Directory.GetFiles(rvt_folder, "*.rvt")

for rvt_file in rvt_files:
    file_name = Path.GetFileName(rvt_file)
    bg_doc = None
    
    try:
        # ตรวจสอบข้อมูลไฟล์ก่อนเปิดว่าเป็น Workshared (Central) หรือไม่
        file_info = BasicFileInfo.Extract(rvt_file)
        is_workshared = file_info.IsWorkshared
        
        model_path = ModelPathUtils.ConvertUserVisiblePathToModelPath(rvt_file)
        open_opts = OpenOptions()
        
        if is_workshared:
            # กรณีที่ 1: เป็น Central File (ทำ Local -> เปิด)
            local_name = file_name.replace(".rvt", "_" + username + ".rvt")
            local_path_str = Path.Combine(local_save_dir, local_name)
            local_path = ModelPathUtils.ConvertUserVisiblePathToModelPath(local_path_str)
            
            WorksharingUtils.CreateNewLocal(model_path, local_path)
            bg_doc = app.OpenDocumentFile(local_path, open_opts)
        else:
            # กรณีที่ 2: เป็นไฟล์ Standalone ธรรมดา (เปิดไฟล์ตรงๆ เลย)
            bg_doc = app.OpenDocumentFile(model_path, open_opts)
        
        # --- เริ่มขั้นตอนการแก้ไข Project Info ---
        t = Transaction(bg_doc, "Update Project Info from Excel")
        t.Start()
        
        project_info = bg_doc.ProjectInformation
        updated_count = 0
        
        for data in excel_data:
            p1 = project_info.LookupParameter(data["param_name"])
            if p1 and not p1.IsReadOnly:
                p1.Set(data["thai_name"])
                
            license_param_name = data["param_name"] + " License No."
            p2 = project_info.LookupParameter(license_param_name)
            if p2 and not p2.IsReadOnly:
                p2.Set(data["license_id"])
                
            updated_count += 1
            
        t.Commit()
        # --------------------------------------
        
        # --- ขั้นตอนการ Save หรือ Sync ---
        if is_workshared:
            # ถ้าเป็น Central ให้ Sync
            transact_opts = TransactWithCentralOptions()
            sync_opts = SynchronizeWithCentralOptions()
            sync_opts.SetRelinquishOptions(RelinquishOptions(True))
            sync_opts.SetSaveLocalBefore(True)
            sync_opts.SetSaveLocalAfter(False)
            
            bg_doc.SynchronizeWithCentral(transact_opts, sync_opts)
            status_text = "(Central File - Synced)"
        else:
            # ถ้าเป็น Standalone ให้ Save ทับไฟล์เดิม
            save_opts = SaveOptions()
            bg_doc.Save(save_opts)
            status_text = "(Standalone File - Saved)"
        
        # ปิดไฟล์
        bg_doc.Close(False)
        
        results.append("สำเร็จ: {} {} - อัปเดต {} ข้อมูล".format(file_name, status_text, updated_count))
        
    except Exception as e:
        # หากเกิด Error ให้พยายามปิดไฟล์ (ถ้าไฟล์เปิดค้างไว้) ก่อนข้ามไปไฟล์ต่อไป
        if bg_doc:
            bg_doc.Close(False)
        results.append("เกิดข้อผิดพลาดกับไฟล์ " + file_name + ": " + str(e))

OUT = results
